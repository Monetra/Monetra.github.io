17/*! \defgroup LM_api Monetra API
18 * Monetra API based on mstdlib. The supported next-generation library.
19 * Use \#include <monetra_api.h>.
20 *
21 * Simple usage example below:
22 *
23 * \code{.c}
24 * #include <monetra_api.h>
25 *
26 * static void trans_callback(LM_conn_t *conn, M_event_t *event, LM_event_type_t type, LM_trans_t *trans)
27 * {
28 * M_list_str_t *params;
29 * size_t i;
30 *
31 * switch (type) {
32 * case LM_EVENT_CONN_CONNECTED:
33 * M_printf("Connected successfully\n");
34 * break;
35 *
36 * case LM_EVENT_CONN_DISCONNECT:
37 * case LM_EVENT_CONN_ERROR:
38 * M_printf("Disconnect or Error received: %s\n", LM_conn_error(conn));
39 * LM_conn_destroy(conn);
40 * M_event_done(event);
41 * break;
42 *
43 * case LM_EVENT_TRANS_DONE:
44 * M_printf("Transaction %p complete:\n", trans);
45 * params = LM_trans_response_keys(trans);
46 * for (i=0; i<M_list_str_len(params); i++) {
47 * const char *key = M_list_str_at(params, i);
48 * M_printf("\t'%s' = '%s'\n", key, LM_trans_response_param(trans, key));
49 * }
50 * M_list_str_destroy(params);
51 * LM_trans_delete(trans);
52 * // We're only running a single transaction, issue a graceful disconnect
53 * LM_conn_disconnect(conn);
54 * break;
55 *
56 * case LM_EVENT_TRANS_TIMEOUT:
57 * M_printf("\tTransaction %p timed out\n", trans);
58 * // Transaction will actually stay enqueued and may later return LM_EVENT_TRANS_DONE.
59 * break;
60 *
61 * case LM_EVENT_TRANS_ERROR:
62 * case LM_EVENT_TRANS_NOCONNECT:
63 * M_printf("Transaction %p error (connectivity): %s\n", trans, LM_conn_error(conn));
64 * LM_trans_delete(trans);
65 * // We should still get an LM_EVENT_CONN_ERROR after this event for additional cleanup.
66 * break;
67 * }
68 * }
69 *
70 * int main(int argc, char **argv)
71 * {
72 * M_event_t *event = M_event_create(M_EVENT_FLAG_NONE);
73 * LM_conn_t *conn = LM_conn_init(event, trans_callback, "testbox.monetra.com", 8665);
74 * LM_trans_t *trans;
75 *
76 * LM_conn_connect(conn);
77 *
78 * trans = LM_trans_new(conn);
79 * LM_trans_set_param(trans, "username", "test_retail:public");
80 * LM_trans_set_param(trans, "password", "publ1ct3st");
81 * LM_trans_set_param(trans, "action", "sale");
82 * LM_trans_set_param(trans, "account", "4012888888881881");
83 * LM_trans_set_param(trans, "expdate", "1220");
84 * LM_trans_set_param(trans, "zip", "32606");
85 * LM_trans_set_param(trans, "cv", "999");
86 * LM_trans_set_param(trans, "amount", "12.00");
87 * LM_trans_send(trans);
88 *
89 * M_event_loop(event, M_TIMEOUT_INF);
90 *
91 * // NOTE: we cleaned up 'conn' within the trans_callback. We can't have
92 * // exited the event loop otherwise.
93 * conn = NULL;
94 *
95 * M_event_destroy(event);
96 *
97 * M_library_cleanup();
98 *
99 * return 0;
100 * }
101 * \endcode
102 *
103 *
104 * \defgroup LM_init Library Initialization
105 * \ingroup LM_api
106 * Data types and functions specific to library initialization.
107 *
108 *
109 * \defgroup LM_conn Connection Management
110 * \ingroup LM_api
111 * Data types and functions related to connection management.
112 *
113 *
114 * \defgroup LM_trans Transaction Management
115 * \ingroup LM_api
116 * Data types and functions related to transaction management.
117 *
118 */
119

143/*! Events triggered. Contains both connection-specific events and transaction-specific
144 * events. */
145enum LM_event_type {
146 /* Communications-related events */
147 LM_EVENT_CONN_CONNECTED = 0x01, /*!< Connection established successfully */
148 LM_EVENT_CONN_DISCONNECT = 0x02, /*!< Graceful remote disconnect. No more events will
149 * arrive after this unless the user requests
150 * connection re-establishment */
151 LM_EVENT_CONN_ERROR = 0x03, /*!< Communications error. No more events will
152 * arrive after this unless the user requests
153 * connection re-establishment */
154 /* Transaction-related events */
155 LM_EVENT_TRANS_DONE = 0x10, /*!< Transaction Complete */
156 LM_EVENT_TRANS_ERROR = 0x11, /*!< Transaction Error (e.g. comms failed). *
157 * Guaranteed to receive a LM_EVENT_CONN_ERROR *
158 * after all transactions are notified */
159 LM_EVENT_TRANS_NOCONNECT = 0x12, /*!< Notification that the transaction is still
160 * ready to be sent, but there was an attempt to
161 * connect that failed. This transaction is still
162 * eligible to be sent if/when a successful connection
163 * is established. Integrators may wish to use this
164 * event to destroy the transaction object if they
165 * consider this a critical failure. */
166 LM_EVENT_TRANS_TIMEOUT = 0x13 /*!< Notification that the user-specified transaction
167 * timeout has elapsed. The user may choose to
168 * ignore this and continue waiting for a final
169 * event, or clean up the transaction. */
170};
171typedef enum LM_event_type LM_event_type_t;

253/*! Callback that is called for each event, both connection-related and transaction related.
254 *
255 * On connection error, each pending transaction will trigger an ERROR event, _followed_
256 * by the connection error or disconnect event itself.
257 *
258 * It is safe to call LM_conn_destroy() or LM_trans_delete() within the callback, however if
259 * there are events still enqueued for a connection when destroyed, they will not be delivered
260 * so it is recommended to only LM_conn_destroy() on LM_EVENT_CONN_DISCONNECT or LM_EVENT_CONN_ERROR.
261 *
262 * It is also possible that if LM_trans_delete() is called in the callback that events
263 * for that transaction may still be delivered for a short time after, but it is guaranteed
264 * that if that occurs, the passed in trans pointer will still be valid as actual destruction
265 * will be delayed, but LM_trans_get_userdata() will return NULL (even if LM_trans_set_userdata()
266 * had previously set it to a valid pointer).
267 *
268 * \param[in] conn Connection handle associated with the event
269 * \param[in] event mstdlib event handle that generated the event
270 * \param[in] type Type of event generated
271 * \param[in] trans Transaction pointer affected if type is LM_EVENT_TRANS_DONE,
272 * LM_EVENT_TRANS_ERROR, or LM_EVENT_TRANS_NOCONECT. NULL if type
273 * is LM_EVENT_CONN_*.
274 */
275typedef void (*LM_event_callback_t)(LM_conn_t *conn, M_event_t *event, LM_event_type_t type, LM_trans_t *trans);

277/*! Initialize a connection object.
278 *
279 * A connection object must be created before any transactions can be created. It requires
280 * an initialized mstdlib event handle to be passed to it, which it will use for all

594/*! Set a user-specified timeout in seconds for notification purposes.
595 *
596 * Sets a notification timeout on the transaction. When the timeout elapses, an
597 * LM_EVENT_TRANS_TIMEOUT event will be delivered. This event does NOT however
598 * cancel the transaction and instead is simply a notification that the user-specified
599 * timer has elapsed. The user may choose to delete the transaction, or continue
600 * waiting for another signal such as LM_EVENT_TRANS_DONE.
601 *
602 * Only transactions that have not had LM_trans_send() called are eligible to be
603 * set for timeout.
604 *
605 * The timeout timer starts once the transaction has been put on the wire, so if
606 * there are errors connecting, the timer will not start until a connection has
607 * been established. The timer is bound to the same event loop as the established
608 * connection so it is guaranteed they will signal using the same thread.
609 *
610 * \param[in] trans Initialized transaction object by LM_trans_new()
611 * \param[in] timeout_s Seconds after sending for the timeout signal to be delivered.
