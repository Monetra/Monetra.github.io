37 /*! \addtogroup m_bitlist Bitwise flags parser and generator
38  * \ingroup mstdlib_base
39  *
40  * Allows for easy creation of a data structure to parse and generate human
41  * readable flags lists made up of bits.
42  *
43  * \code{.c}
44  * static const M_bitlist_t myflags[] = {
45  * { 1 << 0, "flag1" },
46  * { 1 << 1, "flag2" },
47  * { 1 << 2, "flag3" },
48  * { 0, NULL }
49  * };
50  *
51  * M_uint64 initial_flags = (1<<0) | (1<<2);
52  * char *human_flags = NULL;
53  * M_uint64 final_flags = 0;
54  * M_bool rv = M_FALSE;
55  *
56  * if (!M_bitlist_list(&human_flags, M_BITLIST_FLAG_NONE, myflags, initial_flags, '|', error, sizeof(error)))
57  * goto fail;
58  *
59  * if (!M_bitlist_parse(&final_flags, M_BITLIST_FLAG_NONE, myflags, human_flags, '|', error, sizeof(error)))
60  * goto fail;
61  *
62  * if (initial_flags != final_flags) {
63  * M_snprintf(error, sizeof(error), "initial flags don't match converted");
64  * goto fail;
65  * }
66  * rv = M_TRUE;
67  *
68  * fail:
69  * M_free(human_flags);
70  * if (!rv) {
71  * M_printf("FAILURE: %s", error);
72  * }
73  *
74  * \endcode
75  *
76  * @{
77  */
78 
79 /*! Data structure to be created as an array and filled in by caller. Must be
80  * terminated by an entry where the name parameter is set to NULL */
81 typedef struct {
82  M_uint64 id; /*!< The bit to set, usually a power of 2 */
83  const char *name; /*!< Human-readable name associated with the flag/bit */
84 } M_bitlist_t;

86 /*! Flags that may be passed on to the parser or human-readable generator */
87 typedef enum {
88  M_BITLIST_FLAG_NONE = 0, /*!< No Flags */
89  M_BITLIST_FLAG_DONT_TRIM_WHITESPACE = 1<<0, /*!< Parse only. Don't trim whitespace that might surround flags. */
90  M_BITLIST_FLAG_CASE_SENSITIVE = 1<<1, /*!< Parse only. Case sensitive flag matching. */
91  M_BITLIST_FLAG_IGNORE_UNKNOWN = 1<<2, /*!< If a bit is set that is not known, ignore */
92  M_BITLIST_FLAG_DONT_REQUIRE_POWEROF2 = 1<<3 /*!< Don't require a field to be a power of 2. */
93 } M_bitlist_flags_t;
