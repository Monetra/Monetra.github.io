45 /*! Create a DNS resolver handle.
46  *
47  * This resolver handle is responsible for caching DNS results as well as
48  * tracking which associated IP addresses resulted in successful or failed
49  * connections for optimizing future connection attempts. This DNS handle will
50  * be passed into functions that require DNS resolution like
51  * M_io_net_client_create().
52  *
53  * It is recommended to create a single DNS resolver handle at startup and
54  * pass the same handle to all functions which need it, and destroy the handle
55  * at shutdown.
56  *
57  * \return Initialized DNS handle
58  */

72 /*! Set the maximum query time before a timeout is returned.
73  *
74  * In some cases, if a prior result is cached, the query may still return
75  * success rather than a timeout failure at the end of a timeout, but the
76  * result will be stale. If set to 0, will use the internal default of 5000ms
77  * (5s)
78  *
79  * \param[in] dns Initialized DNS object
80  * \param[in] timeout_ms Timeout specified in milliseconds. If provided as 0,

88 /*! Set the maximum amount of time a DNS query can be cached for where results are

124 /*! Returned IP address list is cleaned up immediately after the callback returns, if persistence
125  * is needed, duplicate the list. The list is returned sorted in preference order:
126  * * List starts as alternating between ipv6 and ipv4 addresses in the order returned from
127  * the DNS server, such as ipv6-1, ipv4-1, ipv6-2, ipv4-2 and so on.
128  * * List then is updated with the happyeyeballs RFC6555/8305 status for prior connection
129  * attempts.
130  * * Finally the list is sorted by happyeyeballs status as the primary sort comparison,
131  * followed by the original order as per server preference.
132  * \param[in] ipaddrs Ip addresses resolved in string form, nor NULL if none
133  * \param[in] cb_data Callback data provided to M_dns_gethostbyname()
134  * \param[in] result Result code of DNS lookup
135  */
136 typedef void (*M_dns_ghbn_callback_t)(const M_list_str_t *ipaddrs, void *cb_data, M_dns_result_t result);

139 /*! Request to resolve a DNS hostname to one or more IP addresses. When the resolution
140  * is complete, the callback passed in will be called. This function may call the
141  * callback immediately if the DNS result is cached. Once the supplied callback is
142  * called, the query will be automatically cleaned up.
143  *
144  * \param[in] dns Handle to DNS pointer created with M_dns_create()
145  * \param[in] event Optional. Event handle to use to deliver the result callback
146  * to. This is useful to ensure the result is enqueued to the
147  * same event loop as requested which may limit need for
148  * possible external locking.
149  * \param[in] hostname Hostname to look up.
150  * \param[in] type Type of lookup to perform (IPv4, IPv6, or both/any)
151  * \param[in] callback Callback to call on completion
152  * \param[in] cb_data User data handle passed to callback on completion
153  */
154 M_API void M_dns_gethostbyname(M_dns_t *dns, M_event_t *event, const char *hostname, M_io_net_type_t type, M_dns_ghbn_callback_t callback, void *cb_data);
